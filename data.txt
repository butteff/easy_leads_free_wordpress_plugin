
SYSROLL
{
  BNAME VARCHAR(128),
  BCREATOR VARCHAR(128),
  BTYPE CHAR(1),
  DTBNAME VARCHAR(128),
  DTBCREATOR VARCHAR(128),
  DCONSTNAME VARCHAR(128),
  BMODULEID INTEGER nullable;
  INDCONSTDEP01(unique):     +BCREATOR+BNAME+BMODULEID+BTYPE+DTBCREATOR+DTBNAME+DCONSTNAME;
  INDCONSTDEP02(non-unique): +DTBCREATOR+DTBNAME+DCONSTNAME+BTYPE;
  SCAN_OBJ_SCHEMA(BCREATOR);
}

SYSCLASS
{
  NAME VARCHAR(128),
  TBNAME VARCHAR(128),
  SCHEMA VARCHAR(128),
  TYPE CHAR(1),
  SEQNO SMALLINT,
  VALCOUNT BIGINT updateable,
  DISTCOUNT BIGINT nullable updateable;
  INDCOLDIST01(unique): +SCHEMA+TBNAME+NAME+TYPE+SEQNO;
  SCAN_OBJ_SCHEMA(SCHEMA);
}

SYSMONITOR
{
  NAME VARCHAR(128),
  DEFINER VARCHAR(128),
  TARGET_TYPE CHAR(1),
  TARGET VARCHAR(762),
  MAXFILES INTEGER nullable,
  MAXFILESIZE INTEGER nullable,
  BUFFERSIZE INTEGER nullable,
  DEFINERTYPE CHAR(1),
  EVMONID INTEGER,
  VERSIONNUMBER INTEGER;
  INDEVENTMONITORS01(unique): +NAME;
  INDEVENTMONITORS02(unique): -EVMONID;
}
